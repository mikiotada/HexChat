import {Component, OnInit} from '@angular/core';
import {ActivatedRoute, Router} from '@angular/router';
import {FormBuilder, FormControl, FormGroup, Validators} from '@angular/forms';
import {MatDialogModule} from '@angular/material/dialog';
import {Observable} from "rxjs";
import {map, startWith} from 'rxjs/operators';
import { ViewChild, ChangeDetectionStrategy } from '@angular/core';
// import { AvatarDialogComponent } from "../avatar-dialog/avatar-dialog.component";
import {FirebaseService} from '../../shared/services/firebase.service';
import {AuthService} from "../../shared/services/auth.service";
import * as countries from '../../../assets/Countries.json';

@Component({
    selector: 'app-edit-user',
    templateUrl: './edit-user.component.html',
    styleUrls: ['./edit-user.component.css']
})
export class EditUserComponent implements OnInit {

    exampleForm: FormGroup;
    item: any;
    userId;
    userProfile;

    validation_messages = {
        'name': [
            {type: 'required', message: 'Name is required.'}
        ],

        'age': [
            {type: 'required', message: 'Age is required.'},
        ]
    };

    country = new FormControl();
    options: string[] = countries.countryList;
    filteredOptions: Observable<string[]>;

    constructor(
        public firebaseService: FirebaseService,
        public authService: AuthService,
        private route: ActivatedRoute,
        private fb: FormBuilder,
        private router: Router,
        public dialog: MatDialogModule
    ) {
    }

    ngOnInit() {

        this.filteredOptions = this.country.valueChanges
            .pipe(
                startWith(''),
                map(value => this._filter(value))
            );

        this.authService.user$.subscribe(user => {
            if (user) {
                this.userId = user.uid;
                this.userProfile = user.profile;

                // console.log(user)
                this.createForm();
            }
        });

    }


    createForm() {
        // console.log("this.userProfile:",this.userProfile)
        console.log(this.userProfile.country)
        this.country = this.userProfile.country;
        this.exampleForm = this.fb.group({
            name: [this.userProfile.name, Validators.required],
            age: [this.userProfile.age, Validators.required],
            country: [this.userProfile.country, Validators.required]
        });
    }


    onSubmit(value) {
        // value.avatar = this.item.avatar;
        value.country = value.country;
        console.log(this.country.value)
        value.profileImageUrl = this.userProfile.profileImageUrl;
        value.age = Number(value.age);
        console.log(value);
        this.firebaseService.updateUser(this.userId, value)
            .then(
                res => {
                    // this.router.navigate(['preferences']);
                }
            )
    }

    cancel() {
        this.router.navigate(['preferences']);
    }

    private _filter(value: string): string[] {
        const filterValue = value.toLowerCase();
        console.log("sakshi")
        return this.options.filter(option => option.toLowerCase().includes(filterValue));
    }

}
